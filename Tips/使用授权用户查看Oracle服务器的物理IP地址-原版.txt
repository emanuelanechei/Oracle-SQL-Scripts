acl => 'UTL_INADDR.xml',
description => 'utl_inaddr',
principal => 'DMPMON',  -- DMP监控用户名为大写
is_grant => TRUE,
privilege => 'resolve'
);
commit;
end;
/

host => 'yyds01'  -- 主机名

commit;
end;

end;
