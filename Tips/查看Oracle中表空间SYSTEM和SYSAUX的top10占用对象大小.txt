1. SYSTEM表空间:

SET LINESIZE 300
COLUMN segment_name FORMAT a25
COLUMN segment_type FORMAT a10
COLUMN owner        FORMAT a8

SELECT * FROM
(SELECT segment_name
        , bytes/1024/1024 AS mb
        , segment_type
        , owner
 FROM dba_segments
 WHERE tablespace_name = 'SYSTEM'
 ORDER BY bytes DESC
)
WHERE rownum < 10
/
